# CleverSwitch v0.2 — Dashboard, Historie, Telemetrie, Branding

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Lokale Usage-Historie (JSONL), ein Dashboard-Fenster (Analyse / Accounts / Feedback), Opt-in-Telemetrie über PostHog EU und Clevermation-Branding — alles nativ Swift, null neue Dependencies.

**Architecture:** Zwei neue Kit-Actors (`HistoryStore`, `TelemetryClient`) nach dem KeychainGate-Muster; das Dashboard nutzt den OnboardingWindow-AppKit-Host; Charts über Apples Swift-Charts-Framework. Telemetrie ist pseudonym (Install-UUID), Event-Namen und Property-Keys sind Enums (Compile-Zeit-Whitelist gegen PII). Branding über `Brand.swift`-Token (Flieder `#9458f5`-Familie), gebündelte Plus-Jakarta-Sans-TTFs (SIL OFL) und vorgerenderte Logo-PNGs.

**Tech Stack:** Swift 6 / SwiftUI / Swift Charts (Apple, macOS 13+) / SPM / swift-testing. Keine Drittanbieter-Pakete.

**Architektur-Review:** Blueprint von code-architect-Agent; Codex-Gegencheck eingearbeitet — insbesondere: Actors sind reentrant (Epoch-Guard im Flush), Queue-Rotation nach `inflight-<uuid>.jsonl` statt Ganz-Datei-Löschen, differenzierte HTTP-Status-Behandlung, echtes POSIX `O_APPEND`.

```swift
// Lokale Usage-Historie als JSONL-Monatsdateien unter
// ~/.clevermation/cleverswitch/history/usage-YYYY-MM.jsonl — append-only, eine
// Zeile = ein Sample (5-min-Poll) oder ein Event (Switch/Auto-Switch/Frühwarnung).
// Actor: serialisiert die Datei-IO (Poll, Switch und Dashboard greifen parallel zu).
// Schreiben über POSIX O_APPEND (eine Zeile = ein write()-Syscall = atomar);
// die letzte Zeile darf kaputt sein (Crash mitten im Write) und wird beim Lesen
// still verworfen.

import Foundation

public struct UsageSample: Codable, Sendable, Equatable {
    public let ts: Date
    public let accountId: String
    public let sessionPct: Double
    public let weeklyPct: Double

    public init(ts: Date, accountId: String, sessionPct: Double, weeklyPct: Double) {
        self.ts = ts
        self.accountId = accountId
        self.sessionPct = sessionPct
        self.weeklyPct = weeklyPct
    }
}

public struct HistoryEvent: Codable, Sendable, Equatable {
    public enum Kind: String, Codable, Sendable {
        case manualSwitch = "switch"
        case autoswitch
        case nearLimit = "near_limit"
    }

    public let ts: Date
    public let kind: Kind
    public let provider: String

    public init(ts: Date, kind: Kind, provider: String) {
        self.ts = ts
        self.kind = kind
        self.provider = provider
    }
}

public actor HistoryStore {
    private let baseURL: URL
    private let encoder: JSONEncoder
    private let decoder: JSONDecoder

    public init(baseURL: URL = ClevermationHome.defaultBaseURL()) {
        self.baseURL = baseURL
        encoder = JSONEncoder()
        encoder.dateEncodingStrategy = .iso8601
        decoder = JSONDecoder()
        decoder.dateDecodingStrategy = .iso8601
    }

    private var historyDir: URL { baseURL.appendingPathComponent("history", isDirectory: true) }

    static func monthName(for date: Date) -> String {
        let parts = Calendar(identifier: .gregorian).dateComponents([.year, .month], from: date)
        return String(format: "usage-%04d-%02d.jsonl", parts.year ?? 0, parts.month ?? 0)
    }

    public func appendSample(_ sample: UsageSample) {
        if let data = try? encoder.encode(sample) { appendLine(data, at: sample.ts) }
    }

    public func appendEvent(_ event: HistoryEvent) {
        if let data = try? encoder.encode(event) { appendLine(data, at: event.ts) }
    }

    /// Eine Zeile, ein write()-Syscall: O_APPEND macht das atomar (Zeilen < 200 Byte).
    private func appendLine(_ json: Data, at date: Date) {
        try? ClevermationHome.ensureDirectory(historyDir)
        let path = historyDir.appendingPathComponent(Self.monthName(for: date)).path
        let fd = open(path, O_WRONLY | O_CREAT | O_APPEND, 0o600)
        guard fd >= 0 else { return }  // still verschlucken — das Kit hat bewusst keinen Logger
        var line = json
        line.append(0x0A)
        line.withUnsafeBytes { _ = write(fd, $0.baseAddress, $0.count) }
        close(fd)
    }

    public func loadSamples(monthsBack: Int = 1) -> [UsageSample] {
        lines(monthsBack: monthsBack).compactMap { try? decoder.decode(UsageSample.self, from: $0) }
    }

    public func loadEvents(monthsBack: Int = 1) -> [HistoryEvent] {
        lines(monthsBack: monthsBack).compactMap { try? decoder.decode(HistoryEvent.self, from: $0) }
    }

    private func lines(monthsBack: Int) -> [Data] {
        let calendar = Calendar(identifier: .gregorian)
        var result: [Data] = []
        for offset in stride(from: monthsBack - 1, through: 0, by: -1) {
            guard let month = calendar.date(byAdding: .month, value: -offset, to: Date()) else {
                continue
            }
            let url = historyDir.appendingPathComponent(Self.monthName(for: month))
            guard let raw = try? Data(contentsOf: url) else { continue }
            // kaputte (letzte) Zeilen fallen beim per-Zeile-Decode einfach raus
            result.append(contentsOf: raw.split(separator: 0x0A).map(Data.init))
        }
        return result
    }

    public func pruneOldMonths(keep: Int = 12) {
        guard
            let files = try? FileManager.default.contentsOfDirectory(atPath: historyDir.path)
        else { return }
        let sorted = files.filter { $0.hasPrefix("usage-") && $0.hasSuffix(".jsonl") }.sorted()
        for name in sorted.dropLast(keep) {
            try? FileManager.default.removeItem(
                at: historyDir.appendingPathComponent(name))
        }
    }
}
```

```swift
// Opt-in-Telemetrie (pseudonym) mit Offline-Queue. Sicherheits-Design:
// - Event-Namen und Property-Keys sind ENUMS -> zur Compile-Zeit kann kein
//   Handle/Token/Pfad in die Statistik geraten. Freitext existiert NUR im
//   Feedback-Pfad (sendFeedback), der eine eigene Einwilligung darstellt.
// - Actor + Epoch-Guard: Actors sind reentrant — läuft flush() über ein await,
//   kann setEnabled(false) dazwischen Queue+Identität löschen. Jeder await-
//   Rückweg prüft die Epoche und bricht dann ab.
// - Queue-Rotation: vor dem Senden wird queue.jsonl atomar zu inflight-<uuid>
//   umbenannt; neue Events laufen sofort in eine frische Queue. Erfolg löscht
//   nur die Inflight-Datei -> kein Event geht verloren, keins wird doppelt
//   riskiert (PostHog dedupliziert ohnehin nicht — at-least-once ist ok).
// - Status-Matrix: 2xx löschen · 400/413/422 löschen (Poison-Batch) ·
//   401/403/404/429/5xx/Netzfehler behalten (Retry beim nächsten Poll).

import Foundation

public enum TelemetryEventName: String, Sendable, CaseIterable {
    case appLaunch = "app_launch"
    case switched = "switch"
    case autoSwitched = "auto_switch"
    case loginSuccess = "login_success"
    case loginFailed = "login_failed"
    case nearLimit = "near_limit"
    case appUpdated = "app_updated"
    case cliUpdated = "cli_updated"
    case onboardingDone = "onboarding_done"
    case feedback
}

public enum TelemetryPropertyKey: String, Sendable {
    case provider, mode, reason, trigger
    case appVersion = "app_version"
    case osVersion = "os_version"
    case language
}

public actor TelemetryClient {
    private let http: HTTPClient
    private let baseURL: URL
    private let apiKey: String
    private let endpoint: URL
    private var enabled = false
    private var epoch = 0
    private let encoder: JSONEncoder

    public init(
        http: HTTPClient, baseURL: URL = ClevermationHome.defaultBaseURL(),
        apiKey: String = TelemetryConfig.projectKey,
        endpoint: URL = TelemetryConfig.endpoint
    ) {
        self.http = http
        self.baseURL = baseURL
        self.apiKey = apiKey
        self.endpoint = endpoint
        encoder = JSONEncoder()
        encoder.dateEncodingStrategy = .iso8601
    }

    private var queueURL: URL { baseURL.appendingPathComponent("telemetry-queue.jsonl") }
    private var idURL: URL { baseURL.appendingPathComponent("telemetry-id") }

    public func setEnabled(_ on: Bool) {
        enabled = on
        if !on {
            epoch += 1
            try? FileManager.default.removeItem(at: queueURL)
            try? FileManager.default.removeItem(at: idURL)
            for url in inflightFiles() { try? FileManager.default.removeItem(at: url) }
        }
    }

    public func installId() -> String {
        if let existing = try? String(contentsOf: idURL, encoding: .utf8),
            !existing.isEmpty
        {
            return existing.trimmingCharacters(in: .whitespacesAndNewlines)
        }
        let fresh = UUID().uuidString
        try? ClevermationHome.ensureDirectory(baseURL)
        try? fresh.data(using: .utf8)?.write(to: idURL)
        try? FileManager.default.setAttributes(
            [.posixPermissions: 0o600], ofItemAtPath: idURL.path)
        return fresh
    }

    public func enqueue(
        _ name: TelemetryEventName, _ properties: [TelemetryPropertyKey: String] = [:]
    ) {
        guard enabled else { return }
        var payload: [String: String] = ["distinct_id": installId()]
        for (key, value) in properties { payload[key.rawValue] = value }
        let line: [String: Any] = [
            "event": name.rawValue, "properties": payload,
            "timestamp": ISO8601DateFormatter().string(from: Date()),
        ]
        guard let json = try? JSONSerialization.data(withJSONObject: line) else { return }
        try? ClevermationHome.ensureDirectory(baseURL)
        let fd = open(queueURL.path, O_WRONLY | O_CREAT | O_APPEND, 0o600)
        guard fd >= 0 else { return }
        var data = json
        data.append(0x0A)
        data.withUnsafeBytes { _ = write(fd, $0.baseAddress, $0.count) }
        close(fd)
    }

    private func inflightFiles() -> [URL] {
        ((try? FileManager.default.contentsOfDirectory(
            at: baseURL, includingPropertiesForKeys: nil)) ?? [])
            .filter { $0.lastPathComponent.hasPrefix("inflight-") }
    }

    public func flush() async {
        guard enabled else { return }
        // Queue atomar in eine Inflight-Datei rotieren (neue Events laufen weiter)
        if FileManager.default.fileExists(atPath: queueURL.path) {
            let inflight = baseURL.appendingPathComponent("inflight-\(UUID().uuidString).jsonl")
            try? FileManager.default.moveItem(at: queueURL, to: inflight)
        }
        let startEpoch = epoch
        for file in inflightFiles() {
            guard let raw = try? Data(contentsOf: file) else { continue }
            let events = raw.split(separator: 0x0A).compactMap {
                try? JSONSerialization.jsonObject(with: Data($0))
            }
            guard !events.isEmpty else {
                try? FileManager.default.removeItem(at: file)
                continue
            }
            let body: [String: Any] = ["api_key": apiKey, "batch": events]
            guard let data = try? JSONSerialization.data(withJSONObject: body) else { continue }
            var request = URLRequest(url: endpoint)
            request.httpMethod = "POST"
            request.setValue("application/json", forHTTPHeaderField: "Content-Type")
            request.httpBody = data
            let status = (try? await http.send(request))?.status ?? -1
            // Reentrancy: setEnabled(false) kann während des awaits gelaufen sein
            guard epoch == startEpoch else { return }
            switch status {
            case 200..<300, 400, 413, 422:
                // Erfolg ODER Poison-Batch (würde nie durchgehen) -> weg damit
                try? FileManager.default.removeItem(at: file)
            default:
                // 401/403/404 (Konfig), 429, 5xx, Netzfehler -> behalten, später erneut
                return
            }
        }
    }

    /// Feedback: eigene Einwilligung durch das Absenden selbst; bewusst mit frischer
    /// Zufalls-UUID (nicht Install-ID) -> nicht mit der Statistik verknüpfbar.
    public func sendFeedback(text: String, appVersion: String) async -> Bool {
        let trimmed = String(text.prefix(4_000))
        let event: [String: Any] = [
            "event": TelemetryEventName.feedback.rawValue,
            "properties": [
                "distinct_id": UUID().uuidString,
                "message": trimmed,
                "app_version": appVersion,
            ],
            "timestamp": ISO8601DateFormatter().string(from: Date()),
        ]
        let body: [String: Any] = ["api_key": apiKey, "batch": [event]]
        guard let data = try? JSONSerialization.data(withJSONObject: body) else { return false }
        var request = URLRequest(url: endpoint)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = data
        let status = (try? await http.send(request))?.status ?? -1
        return (200..<300).contains(status)
    }
}
```

                    metricCard(
                        L10n.t("dashboard_metric_avoided"),
                        value: HistoryAnalytics.avoidedStops(events: events, since: weekAgo),
                        accent: true)
                }
                ForEach(model.state.accounts, id: \.id) { account in
                    accountChart(for: account)
                }
                if samples.isEmpty {
                    Text(L10n.t("dashboard_no_data"))
                        .font(.callout)
                        .foregroundStyle(.secondary)
                        .frame(maxWidth: .infinity, alignment: .center)
                        .padding(.top, 30)
                }
            }
            .padding(18)
        }
        .task {
            events = await model.loadHistoryEvents(monthsBack: 2)
            samples = await model.loadHistorySamples(monthsBack: 2)
        }
    }

    private func metricCard(_ title: String, value: Int, accent: Bool = false) -> some View {
        VStack(alignment: .leading, spacing: 4) {
            Text(title).font(.caption).foregroundStyle(.secondary)
            Text("\(value)")
                .font(Brand.display(24))
                .foregroundStyle(accent ? Brand.flieder : .primary)
        }
        .frame(maxWidth: .infinity, alignment: .leading)
        .padding(12)
        .background(RoundedRectangle(cornerRadius: 10).fill(.quaternary.opacity(0.5)))
    }

    @ViewBuilder
    private func accountChart(for account: Account) -> some View {
        let data = HistoryAnalytics.samples(samples, accountId: account.id, since: weekAgo)
        if !data.isEmpty {
            VStack(alignment: .leading, spacing: 6) {
                Text(model.displayHandle(account.handle))
                    .font(.callout.weight(.medium))
                Chart(data, id: \.ts) { sample in
                    LineMark(
                        x: .value("Zeit", sample.ts),
                        y: .value("7d %", sample.weeklyPct))
                    .foregroundStyle(Brand.flieder)
                    LineMark(
                        x: .value("Zeit", sample.ts),
                        y: .value("Session %", sample.sessionPct))
                    .foregroundStyle(Brand.korall)
                    .opacity(0.7)
                }
                .chartYScale(domain: 0...100)
                .frame(height: 110)
            }
        }
    }
}

            Spacer()
        }
        .padding(18)

## Risiken & Wachsamkeitspunkte (aus Architekt- und Codex-Review)

1. **Actor-Reentrancy** ist die Bug-Quelle Nr. 1: nach JEDEM `await` in `TelemetryClient.flush()` die Epoche prüfen (im Code oben drin — beim Erweitern nicht entfernen). Kein fire-and-forget `Task.detached` innerhalb von Actor-Methoden.
2. **Charts-Import**: bei Link-Fehlern im executableTarget nicht lange debuggen — Canvas-Fallback (Path durch die Sample-Punkte, 0–100 normiert) und weiter.
3. **PostHog-Key public** im Binary ist by design — in TELEMETRY.md und Code-Kommentar erklären, sonst kommt das erste GitHub-Issue garantiert dazu.
4. **Feedback-Wording**: Statistik-Toggle aus ≠ „keine Netzwerk-Requests" — das Feedback-Formular sendet trotzdem (eigene Einwilligung). `feedback_hint` kommuniziert das.
5. **MainActor sauber halten**: alles Datei-/HTTP-IO läuft auf den Actor-Executors bzw. detached — nie synchron im Menü-Code.
6. **L10n-Versprechen**: 19 Keys × 16 Sprachen — Übersetzungs-Agent mit Verifikations-Skript ist Pflicht, sonst bricht das README-Versprechen.
